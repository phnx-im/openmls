21/// Contains the messages that are produced by committing.
22#[derive(Debug, Clone)]
23pub struct DmlsCommitMessageBundle {
24    /// The DMLs message that contains the epoch and the MLS message.
25    pub dmls_message: DmlsMessageOut,
26    /// The welcome message that is produced by the commit.
27    pub welcome: Option<Welcome>,
28    /// The group info that is produced by the commit.
29    pub group_info: Option<GroupInfo>,
30}
31
32/// Error processing a DMLs message.
33#[derive(Debug, Error)]
34pub enum ProcessDmlsMessageError<StorageError> {
35    /// DMLS message does not contain Public or Private MLS message.
36    #[error("DMLS message does not contain Public or Private MLS message.")]
37    IncompatibleMessageType,
38    /// Error loading DMLS group state.
39    #[error("Error loading DMLS group state: {0}")]
40    StorageError(StorageError),
41    /// Error processing the MLS message.
42    #[error("Error processing MLS message: {0}")]
43    ProcessMessageError(#[from] ProcessMessageError),
44}
45
46impl DmlsGroup {
47    /// Add members to the group.
48    pub fn add_members<Provider: OpenDmlsProvider>(
49        &mut self,
50        provider: &Provider,
51        signer: &impl Signer,
52        key_packages: &[KeyPackage],
53    ) -> Result<
54        (DmlsMessageOut, MlsMessageOut, Option<GroupInfo>),
55        AddMembersError<<Provider as OpenMlsProvider>::StorageError>,
56    > {
57        let epoch = self.derive_epoch_id(provider.crypto()).unwrap();
58        let provider = provider.provider_for_epoch(epoch.clone());
59        let (mls_message, welcome, group_info) =
60            self.0.add_members(&provider, signer, key_packages)?;
61        let dmls_message = DmlsMessageOut {
62            epoch,
63            message: mls_message,
64        };
65        Ok((dmls_message, welcome, group_info))
66    }
67
68    /// Load the correct group and process a message.
69    pub fn process_message<Provider: OpenDmlsProvider>(
70        &mut self,
71        provider: &Provider,
72        message: DmlsMessageIn,
73    ) -> Result<
74        ProcessedMessage,
75        ProcessDmlsMessageError<<Provider as OpenMlsProvider>::StorageError>,
76    > {
77        let DmlsMessageIn { epoch, message } = message;
78        let protocol_message: ProtocolMessage = match message.body {
79            MlsMessageBodyIn::PublicMessage(public_message_in) => public_message_in.into(),
80            MlsMessageBodyIn::PrivateMessage(private_message_in) => private_message_in.into(),
81            _ => {
82                return Err(ProcessDmlsMessageError::IncompatibleMessageType);
83            }
84        };
85        let provider = provider.provider_for_epoch(epoch);
86        Ok(self.0.process_message(&provider, protocol_message)?)
87    }
88
89    /// DMLS wrapper around the [`MlsGroup::self_update`] function.
90    pub fn self_update<Provider: OpenDmlsProvider>(
91        &mut self,
92        provider: &Provider,
93        signer: &impl Signer,
94        leaf_node_parameters: LeafNodeParameters,
95    ) -> Result<DmlsCommitMessageBundle, SelfUpdateError<<Provider as OpenMlsProvider>::StorageError>>
96    {
97        let epoch = self.derive_epoch_id(provider.crypto()).unwrap();
98        let provider = provider.provider_for_epoch(epoch.clone());
99        let (message, welcome, group_info) = self
100            .0
101            .self_update(&provider, signer, leaf_node_parameters)?
102            .into_contents();
103        let dmls_message = DmlsMessageOut { epoch, message };
104        Ok(DmlsCommitMessageBundle {
105            dmls_message,
106            welcome,
107            group_info,
108        })
109    }
110
111    /// Clear a pending commit
112    pub fn clear_pending_commit<Provider: OpenDmlsProvider>(
113        &mut self,
114        provider: &Provider,
115    ) -> Result<(), <Provider as OpenMlsProvider>::StorageError> {
116        let epoch = self.derive_epoch_id(provider.crypto()).unwrap();
117        let provider = provider.provider_for_epoch(epoch.clone());
118        self.0.clear_pending_commit(provider.storage())?;
119        Ok(())
120    }
121}
