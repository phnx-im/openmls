1//! This module contains the DMLsMessageIn and DMLsMessageOut structs, both wrappers around
2//! MlsMessageIn and MlsMessageOut respectively. Each wrapper also contains an epoch field.
3
4use std::ops::Deref;
5
6use openmls_traits::dmls_traits::DmlsEpoch;
7use tls_codec::{TlsDeserialize, TlsDeserializeBytes, TlsSerialize, TlsSize};
8
9use crate::{
10    framing::{MlsMessageBodyIn, MlsMessageIn, MlsMessageOut},
11    group::GroupId,
12};
13
14/// The [`DmlsMessageIn`] struct is a wrapper around [`MlsMessageIn`] that contains
15/// an additional epoch field.
16#[derive(PartialEq, Debug, Clone, TlsSize, TlsDeserialize, TlsDeserializeBytes)]
17#[cfg_attr(feature = "test-utils", derive(TlsSerialize))]
18pub struct DmlsMessageIn {
19    /// The epoch of the message.
20    pub epoch: DmlsEpoch,
21    /// The actual message.
22    pub message: MlsMessageIn,
23}
24
25impl DmlsMessageIn {
26    /// Returns the epoch of the message.
27    pub fn epoch(&self) -> &DmlsEpoch {
28        &self.epoch
29    }
30
31    /// Returns the group ID of the message.
32    pub fn group_id(&self) -> &GroupId {
33        match &self.message.body {
34            MlsMessageBodyIn::PublicMessage(msg) => msg.group_id(),
35            MlsMessageBodyIn::PrivateMessage(msg) => msg.group_id(),
36            _ => panic!("Invalid message type for group ID extraction"),
37        }
38    }
39}
40
41impl Deref for DmlsMessageIn {
42    type Target = MlsMessageIn;
43
44    fn deref(&self) -> &Self::Target {
45        &self.message
46    }
47}
48
49/// The [`DmlsMessageOut`] struct is a wrapper around [`MlsMessageOut`] that contains
50/// an additional epoch field.
51#[derive(Debug, Clone, PartialEq, TlsSerialize, TlsSize)]
52pub struct DmlsMessageOut {
53    pub(super) epoch: DmlsEpoch,
54    pub(super) message: MlsMessageOut,
55}
56
57impl DmlsMessageOut {
58    /// Returns the epoch of the message.
59    pub fn epoch(&self) -> &DmlsEpoch {
60        &self.epoch
61    }
62}
63
64impl Deref for DmlsMessageOut {
65    type Target = MlsMessageOut;
66
67    fn deref(&self) -> &Self::Target {
68        &self.message
69    }
70}
