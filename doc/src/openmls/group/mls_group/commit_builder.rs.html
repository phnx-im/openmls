84/// The [`CommitBuilder`] is used to easily and dynamically build commit messages.
85/// It operates in a series of stages:
86///
87/// The [`Initial`] stage is used to populate the builder with proposals and other data using
88/// method calls on the builder that let the builder stay in the same stage.
89///
90/// The next stage is [`LoadedPsks`], and it signifies the stage after the builder loaded the the
91/// pre-shared keys for the PreSharedKey proposals in this commit.
92///
93/// Then comes the [`Complete`] stage, which denotes that all data has been validated. From this
94/// stage, the commit can be staged in the group, and the outgoing messages returned.
95///
96/// For example, to create a commit to a new Add proposal with a KeyPackage `key_package_to_add`
97/// that does not commit to the proposals in the proposal store, one could build the commit as
98/// follows:
99///
100/// ```rust,ignore
101/// let message_bundle: CommitMessageBundle = mls_group
102///   .commit_builder()
103///   .consume_proposal_store(false)
104///   .add_proposal(key_package_to_add)
105///   .load_psks(provider.storage())?
106///   .build(provider.rand(), provider.crypto(), signer, app_policy_proposals)?
107///   .stage_commit(provider)?;
108///
109/// let commit = message_bundle.commit();
110/// let welcome = message_bundle.welcome().expect("expected a welcome since there was an add");
111/// let group_info = message_bundle.welcome().expect("expected a group info since there was an add");
112/// ```
113///
114/// In this example `signer` is a reference to a [`Signer`] and `app_policy_proposals` is the
115/// application-defined policy for which proposals to accept, implemented by an
116/// `FnMut(&QueuedProposal) -> bool`.
117///
118/// See the [book] for another example.
119///
120/// [book]: https://book.openmls.tech/user_manual/add_members.html
121#[derive(Debug)]
122pub struct CommitBuilder<'a, T> {
123    /// A mutable reference to the MlsGroup. This means that we hold an exclusive lock on the group
124    /// for the lifetime of this builder.
125    group: &'a mut MlsGroup,
126
127    /// The current stage
128    stage: T,
129}
130
131impl<'a, T> CommitBuilder<'a, T> {
132    pub(crate) fn replace_stage<NextStage>(
133        self,
134        next_stage: NextStage,
135    ) -> (T, CommitBuilder<'a, NextStage>) {
136        self.map_stage(|prev_stage| (prev_stage, next_stage))
137    }
138
139    pub(crate) fn into_stage<NextStage>(
140        self,
141        next_stage: NextStage,
142    ) -> CommitBuilder<'a, NextStage> {
143        self.replace_stage(next_stage).1
144    }
145
146    pub(crate) fn take_stage(self) -> (T, CommitBuilder<'a, ()>) {
147        self.replace_stage(())
148    }
149
150    pub(crate) fn map_stage<NextStage, Aux, F: FnOnce(T) -> (Aux, NextStage)>(
151        self,
152        f: F,
153    ) -> (Aux, CommitBuilder<'a, NextStage>) {
154        let Self { group, stage } = self;
155
156        let (aux, stage) = f(stage);
157
158        (aux, CommitBuilder { group, stage })
159    }
160
161    #[cfg(feature = "fork-resolution")]
162    pub(crate) fn stage(&self) -> &T {
163        &self.stage
164    }
165}

174impl<'a> CommitBuilder<'a, Initial> {
175    /// returns a new [`CommitBuilder`] for the given [`MlsGroup`].
176    pub fn new(group: &'a mut MlsGroup) -> Self {
177        let stage = Initial {
178            create_group_info: group.configuration().use_ratchet_tree_extension,
179            ..Default::default()
180        };
181        Self { group, stage }
182    }
183
184    /// Sets whether or not the proposals in the proposal store of the group should be included in
185    /// the commit. Defaults to `true`.
186    pub fn consume_proposal_store(mut self, consume_proposal_store: bool) -> Self {
187        self.stage.consume_proposal_store = consume_proposal_store;
188        self
189    }
190
191    /// Sets whether or not a [`GroupInfo`] should be created when the commit is staged. Defaults to
192    /// the value of the [`MlsGroup`]s [`MlsGroupJoinConfig`].
193    pub fn create_group_info(mut self, create_group_info: bool) -> Self {
194        self.stage.create_group_info = create_group_info;
195        self
196    }
197
198    /// Sets whether or not the commit should force a self-update. Defaults to `false`.
199    pub fn force_self_update(mut self, force_self_update: bool) -> Self {
200        self.stage.force_self_update = force_self_update;
201        self
202    }
203
204    /// Adds a proposal to the proposals to be committed.
205    pub fn add_proposal(mut self, proposal: Proposal) -> Self {
206        self.stage.own_proposals.push(proposal);
207        self
208    }
209
210    /// Adds the proposals in the iterator to the proposals to be committed.
211    pub fn add_proposals(mut self, proposals: impl IntoIterator<Item = Proposal>) -> Self {
212        self.stage.own_proposals.extend(proposals);
213        self
214    }
215
216    /// Sets the leaf node parameters for the new leaf node in a self-update. Implies that a
217    /// self-update takes place.
218    pub fn leaf_node_parameters(mut self, leaf_node_parameters: LeafNodeParameters) -> Self {
219        self.stage.leaf_node_parameters = leaf_node_parameters;
220        self
221    }
222
223    /// Adds an Add proposal to the provided [`KeyPackage`] to the list of proposals to be
224    /// committed.
225    pub fn propose_adds(mut self, key_packages: impl IntoIterator<Item = KeyPackage>) -> Self {
226        self.stage.own_proposals.extend(
227            key_packages
228                .into_iter()
229                .map(|key_package| Proposal::Add(AddProposal { key_package })),
230        );
231        self
232    }
233
234    pub fn propose_removals(mut self, removed: impl IntoIterator<Item = LeafNodeIndex>) -> Self {
235        self.stage.own_proposals.extend(
236            removed
237                .into_iter()
238                .map(|removed| Proposal::Remove(RemoveProposal { removed })),
239        );
240        self
241    }
242
243    pub fn propose_group_context_extensions(mut self, extensions: Extensions) -> Self {
244        self.stage
245            .own_proposals
246            .push(Proposal::GroupContextExtensions(
247                GroupContextExtensionProposal::new(extensions),
248            ));
249        self
250    }
251
252    /// Loads the PSKs for the PskProposals marked for inclusion and moves on to the next phase.
253    pub fn load_psks<Storage: StorageProvider>(
254        self,
255        storage: &'a Storage,
256    ) -> Result<CommitBuilder<'a, LoadedPsks>, CreateCommitError> {
257        let psk_ids: Vec<_> = self
258            .stage
259            .own_proposals
260            .iter()
261            .chain(
262                self.group
263                    .proposal_store()
264                    .proposals()
265                    .map(|queued_proposal| queued_proposal.proposal()),
266            )
267            .filter_map(|proposal| match proposal {
268                Proposal::PreSharedKey(psk_proposal) => Some(psk_proposal.clone().into_psk_id()),
269                _ => None,
270            })
271            .collect();
272
273        // Load the PSKs and make the PskIds owned.
274        let psks = load_psks(storage, &self.group.resumption_psk_store, &psk_ids)?
275            .into_iter()
276            .map(|(psk_id_ref, key)| (psk_id_ref.clone(), key))
277            .collect();
278
279        Ok(self
280            .map_stage(|stage| {
281                (
282                    (),
283                    LoadedPsks {
284                        own_proposals: stage.own_proposals,
285                        psks,
286                        force_self_update: stage.force_self_update,
287                        leaf_node_parameters: stage.leaf_node_parameters,
288                        consume_proposal_store: stage.consume_proposal_store,
289                        create_group_info: stage.create_group_info,
290                    },
291                )
292            })
293            .1)
294    }
295}

297impl<'a> CommitBuilder<'a, LoadedPsks> {
298    /// Validates the inputs and builds the commit. The last argument `f` is a function that lets
299    /// the caller filter the proposals that are considered for inclusion. This provides a way for
300    /// the application to enforce custom policies in the creation of commits.
301    pub fn build<S: Signer>(
302        self,
303        rand: &impl OpenMlsRand,
304        crypto: &impl OpenMlsCrypto,
305        signer: &S,
306        f: impl FnMut(&QueuedProposal) -> bool,
307    ) -> Result<CommitBuilder<'a, Complete>, CreateCommitError> {
308        self.build_internal(rand, crypto, signer, None::<NewSignerBundle<'_, S>>, f)
309    }
310
311    /// Just like `build`, this function validates the inputs and builds the
312    /// commit. The last argument `f` is a function that lets the caller filter
313    /// the proposals that are considered for inclusion. This provides a way for
314    /// the application to enforce custom policies in the creation of commits.
315    ///
316    /// In contrast to `build`, this function can be used to create commits that
317    /// rotate the own leaf node's signature key.
318    pub fn build_with_new_signer<S: Signer>(
319        self,
320        rand: &impl OpenMlsRand,
321        crypto: &impl OpenMlsCrypto,
322        old_signer: &impl Signer,
323        new_signer: NewSignerBundle<'_, S>,
324        f: impl FnMut(&QueuedProposal) -> bool,
325    ) -> Result<CommitBuilder<'a, Complete>, CreateCommitError> {
326        self.build_internal(rand, crypto, old_signer, Some(new_signer), f)
327    }
328
329    fn build_internal<S: Signer>(
330        self,
331        rand: &impl OpenMlsRand,
332        crypto: &impl OpenMlsCrypto,
333        old_signer: &impl Signer,
334        new_signer: Option<NewSignerBundle<'_, S>>,
335        f: impl FnMut(&QueuedProposal) -> bool,
336    ) -> Result<CommitBuilder<'a, Complete>, CreateCommitError> {
337        let ciphersuite = self.group.ciphersuite();
338        let sender = Sender::build_member(self.group.own_leaf_index());
339        let (mut cur_stage, builder) = self.take_stage();
340        let psks = cur_stage.psks;
341
342        // put the pending and uniform proposals into a uniform shape,
343        // i.e. produce queued proposals from the own proposals
344        let own_proposals: Vec<_> = cur_stage
345            .own_proposals
346            .into_iter()
347            .map(|proposal| {
348                QueuedProposal::from_proposal_and_sender(ciphersuite, crypto, proposal, &sender)
349            })
350            .collect::<Result<_, _>>()?;
351
352        // prepare an iterator for the proposals in the group's proposal store, but only if the
353        // flag is set.
354        let group_proposal_store_queue = builder
355            .group
356            .pending_proposals()
357            .filter(|_| cur_stage.consume_proposal_store)
358            .cloned();
359
360        // prepare the iterator for the proposal validation and seletion function. That function
361        // assumes that "earlier in the list" means "older", so since our own proposals are
362        // newest, we have to put them last.
363        let proposal_queue = group_proposal_store_queue.chain(own_proposals).filter(f);
364
365        let (proposal_queue, contains_own_updates) =
366            ProposalQueue::filter_proposals_without_inline(
367                proposal_queue,
368                builder.group.own_leaf_index,
369            )
370            .map_err(|e| match e {
371                ProposalQueueError::LibraryError(e) => e.into(),
372                ProposalQueueError::ProposalNotFound => CreateCommitError::MissingProposal,
373                ProposalQueueError::UpdateFromExternalSender
374                | ProposalQueueError::SelfRemoveFromNonMember => {
375                    CreateCommitError::WrongProposalSenderType
376                }
377            })?;
378
379        // Validate the proposals by doing the following checks:
380
381        // ValSem113: All Proposals: The proposal type must be supported by all
382        // members of the group
383        builder
384            .group
385            .public_group
386            .validate_proposal_type_support(&proposal_queue)?;
387        // ValSem101
388        // ValSem102
389        // ValSem103
390        // ValSem104
391        builder
392            .group
393            .public_group
394            .validate_key_uniqueness(&proposal_queue, None)?;
395        // ValSem105
396        builder
397            .group
398            .public_group
399            .validate_add_proposals(&proposal_queue)?;
400        // ValSem106
401        // ValSem109
402        builder
403            .group
404            .public_group
405            .validate_capabilities(&proposal_queue)?;
406        // ValSem107
407        // ValSem108
408        builder
409            .group
410            .public_group
411            .validate_remove_proposals(&proposal_queue)?;
412        builder
413            .group
414            .public_group
415            .validate_pre_shared_key_proposals(&proposal_queue)?;
416        // Validate update proposals for member commits
417        // ValSem110
418        // ValSem111
419        // ValSem112
420        builder
421            .group
422            .public_group
423            .validate_update_proposals(&proposal_queue, builder.group.own_leaf_index())?;
424
425        // ValSem208
426        // ValSem209
427        builder
428            .group
429            .public_group
430            .validate_group_context_extensions_proposal(&proposal_queue)?;
431
432        let ciphersuite = builder.group.ciphersuite();
433        let sender = Sender::build_member(builder.group.own_leaf_index());
434        let proposal_reference_list = proposal_queue.commit_list();
435
436        // Make a copy of the public group to apply proposals safely
437        let mut diff = builder.group.public_group.empty_diff();
438
439        // Apply proposals to tree
440        let apply_proposals_values =
441            diff.apply_proposals(&proposal_queue, builder.group.own_leaf_index())?;
442        if apply_proposals_values.self_removed {
443            return Err(CreateCommitError::CannotRemoveSelf);
444        }
445
446        let path_computation_result =
447            // If path is needed, compute path values
448            if apply_proposals_values.path_required
449                || contains_own_updates
450                || cur_stage.force_self_update
451                || !cur_stage.leaf_node_parameters.is_empty()
452            {
453                // Process the path. This includes updating the provisional
454                // group context by updating the epoch and computing the new
455                // tree hash.
456                if let Some(new_signer) = new_signer {
457                    if let Some(credential_with_key) =
458                        cur_stage.leaf_node_parameters.credential_with_key()
459                    {
460                        if credential_with_key != &new_signer.credential_with_key {
461                            return Err(CreateCommitError::InvalidLeafNodeParameters);
462                        }
463                    }
464                    cur_stage.leaf_node_parameters.set_credential_with_key(
465                        new_signer.credential_with_key,
466                    );
467                    diff.compute_path(
468                        rand,
469                        crypto,
470                        builder.group.own_leaf_index(),
471                        apply_proposals_values.exclusion_list(),
472                        &CommitType::Member,
473                        &cur_stage.leaf_node_parameters,
474                        new_signer.signer,
475                        apply_proposals_values.extensions.clone()
476                    )?
477                } else {
478                    diff.compute_path(
479                        rand,
480                        crypto,
481                        builder.group.own_leaf_index(),
482                        apply_proposals_values.exclusion_list(),
483                        &CommitType::Member,
484                        &cur_stage.leaf_node_parameters,
485                        old_signer,
486                        apply_proposals_values.extensions.clone()
487                    )?
488                }
489            } else {
490                // If path is not needed, update the group context and return
491                // empty path processing results
492                diff.update_group_context(crypto, apply_proposals_values.extensions.clone())?;
493                PathComputationResult::default()
494            };
495
496        let update_path_leaf_node = path_computation_result
497            .encrypted_path
498            .as_ref()
499            .map(|path| path.leaf_node().clone());
500
501        // Create commit message
502        let commit = Commit {
503            proposals: proposal_reference_list,
504            path: path_computation_result.encrypted_path,
505        };
506
507        // Build AuthenticatedContent
508        let mut authenticated_content = AuthenticatedContent::commit(
509            builder.group.framing_parameters(),
510            sender,
511            commit,
512            builder.group.public_group.group_context(),
513            old_signer,
514        )?;
515
516        // Update the confirmed transcript hash using the commit we just created.
517        diff.update_confirmed_transcript_hash(crypto, &authenticated_content)?;
518
519        let serialized_provisional_group_context = diff
520            .group_context()
521            .tls_serialize_detached()
522            .map_err(LibraryError::missing_bound_check)?;
523
524        let base_commit_secret = path_computation_result
525            .base_commit_secret
526            .unwrap_or_else(|| BaseCommitSecret::zero_secret(ciphersuite));
527        let commit_secret = CommitSecret::new(crypto, ciphersuite, &base_commit_secret)
528            .map_err(LibraryError::unexpected_crypto_error)?;
529
530        let commit_confirmation = CommitConfirmation::new(crypto, ciphersuite, &base_commit_secret)
531            .map_err(LibraryError::unexpected_crypto_error)?;
532
533        let mut old_init_secret = builder.group.group_epoch_secrets.init_secret().clone();
534        let child_init_secret = ChildInitSecret::new(
535            crypto,
536            ciphersuite,
537            &mut old_init_secret,
538            &commit_confirmation,
539            &serialized_provisional_group_context,
540        )
541        .map_err(LibraryError::unexpected_crypto_error)?;
542
543        let joiner_secret = JoinerSecret::new(
544            crypto,
545            ciphersuite,
546            commit_secret,
547            &child_init_secret,
548            &serialized_provisional_group_context,
549        )
550        .map_err(LibraryError::unexpected_crypto_error)?;
551
552        // Prepare the PskSecret
553        let psk_secret = { PskSecret::new(crypto, ciphersuite, psks)? };
554
555        // Create key schedule
556        let mut key_schedule = KeySchedule::init(ciphersuite, crypto, &joiner_secret, psk_secret)?;
557
558        let serialized_provisional_group_context = diff
559            .group_context()
560            .tls_serialize_detached()
561            .map_err(LibraryError::missing_bound_check)?;
562
563        let welcome_secret = key_schedule
564            .welcome(crypto, builder.group.ciphersuite())
565            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
566        key_schedule
567            .add_context(crypto, &serialized_provisional_group_context)
568            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
569        let provisional_epoch_secrets = key_schedule
570            .epoch_secrets(crypto, builder.group.ciphersuite())
571            .map_err(|_| LibraryError::custom("Using the key schedule in the wrong state"))?;
572
573        // Calculate the confirmation tag
574        let confirmation_tag = provisional_epoch_secrets
575            .confirmation_key()
576            .tag(
577                crypto,
578                builder.group.ciphersuite(),
579                diff.group_context().confirmed_transcript_hash(),
580            )
581            .map_err(LibraryError::unexpected_crypto_error)?;
582
583        // Set the confirmation tag
584        authenticated_content.set_confirmation_tag(confirmation_tag.clone());
585
586        diff.update_interim_transcript_hash(ciphersuite, crypto, confirmation_tag.clone())?;
587
588        // If there are invitations, we need to build a welcome
589        let needs_welcome = !apply_proposals_values.invitation_list.is_empty();
590
591        // We need a GroupInfo if we need to build a Welcome, or if
592        // `create_group_info` is set to `true`. If not overridden, `create_group_info`
593        // is set to the `use_ratchet_tree` flag in the group configuration.
594        let needs_group_info = needs_welcome || cur_stage.create_group_info;
595
596        let group_info = if !needs_group_info {
597            None
598        } else {
599            // Build ExternalPub extension
600            let external_pub = provisional_epoch_secrets
601                .external_secret()
602                .derive_external_keypair(crypto, ciphersuite)
603                .map_err(LibraryError::unexpected_crypto_error)?
604                .public;
605            let external_pub_extension =
606                Extension::ExternalPub(ExternalPubExtension::new(external_pub.into()));
607
608            // Create the ratchet tree extension if necessary
609            let extensions: Extensions = if builder.group.configuration().use_ratchet_tree_extension
610            {
611                Extensions::from_vec(vec![
612                    Extension::RatchetTree(RatchetTreeExtension::new(diff.export_ratchet_tree())),
613                    external_pub_extension,
614                ])?
615            } else {
616                Extensions::single(external_pub_extension)
617            };
618
619            // Create to-be-signed group info.
620            let group_info_tbs = {
621                GroupInfoTBS::new(
622                    diff.group_context().clone(),
623                    extensions,
624                    confirmation_tag,
625                    builder.group.own_leaf_index(),
626                )
627            };
628            // Sign to-be-signed group info.
629            Some(group_info_tbs.sign(old_signer)?)
630        };
631
632        let welcome_option = if !needs_welcome {
633            None
634        } else {
635            // Encrypt GroupInfo object
636            let (welcome_key, welcome_nonce) = welcome_secret
637                .derive_welcome_key_nonce(crypto, builder.group.ciphersuite())
638                .map_err(LibraryError::unexpected_crypto_error)?;
639            let encrypted_group_info = welcome_key
640                .aead_seal(
641                    crypto,
642                    group_info
643                        .as_ref()
644                        .ok_or_else(|| LibraryError::custom("GroupInfo was not computed"))?
645                        .tls_serialize_detached()
646                        .map_err(LibraryError::missing_bound_check)?
647                        .as_slice(),
648                    &[],
649                    &welcome_nonce,
650                )
651                .map_err(LibraryError::unexpected_crypto_error)?;
652
653            // Create group secrets for later use, so we can afterwards consume the
654            // `joiner_secret`.
655            let encrypted_secrets = diff.encrypt_group_secrets(
656                &joiner_secret,
657                apply_proposals_values.invitation_list,
658                path_computation_result.plain_path.as_deref(),
659                &apply_proposals_values.presharedkeys,
660                &encrypted_group_info,
661                crypto,
662                builder.group.own_leaf_index(),
663            )?;
664
665            // Create welcome message
666            let welcome = Welcome::new(ciphersuite, encrypted_secrets, encrypted_group_info);
667            Some(welcome)
668        };
669
670        let (provisional_group_epoch_secrets, provisional_message_secrets) =
671            provisional_epoch_secrets.split_secrets(
672                serialized_provisional_group_context,
673                diff.tree_size(),
674                builder.group.own_leaf_index(),
675            );
676
677        let staged_commit_state = MemberStagedCommitState::new(
678            old_init_secret,
679            provisional_group_epoch_secrets,
680            provisional_message_secrets,
681            diff.into_staged_diff(crypto, ciphersuite)?,
682            path_computation_result.new_keypairs,
683            // The committer is not allowed to include their own update
684            // proposal, so there is no extra keypair to store here.
685            None,
686            update_path_leaf_node,
687        );
688        let staged_commit = StagedCommit::new(
689            proposal_queue,
690            StagedCommitState::GroupMember(Box::new(staged_commit_state)),
691        );
692
693        Ok(builder.into_stage(Complete {
694            result: CreateCommitResult {
695                commit: authenticated_content,
696                welcome_option,
697                staged_commit,
698                group_info: group_info.filter(|_| cur_stage.create_group_info),
699            },
700        }))
701    }
702}
703
704impl CommitBuilder<'_, Complete> {
705    #[cfg(test)]
706    pub(crate) fn commit_result(self) -> CreateCommitResult {
707        self.stage.result
708    }
709
710    /// Stages the commit and returns the protocol messages.
711    pub fn stage_commit<Provider: OpenMlsProvider>(
712        self,
713        provider: &Provider,
714    ) -> Result<CommitMessageBundle, CommitBuilderStageError<Provider::StorageError>> {
715        let Self {
716            group,
717            stage: Complete {
718                result: create_commit_result,
719            },
720            ..
721        } = self;
722
723        // Set the current group state to [`MlsGroupState::PendingCommit`],
724        // storing the current [`StagedCommit`] from the commit results
725        group.group_state = MlsGroupState::PendingCommit(Box::new(PendingCommitState::Member(
726            create_commit_result.staged_commit,
727        )));
728
729        provider
730            .storage()
731            .write_group_state(group.group_id(), &group.group_state)
732            .map_err(CommitBuilderStageError::KeyStoreError)?;
733
734        group.reset_aad();
735
736        // Convert PublicMessage messages to MLSMessage and encrypt them if required by the
737        // configuration.
738        //
739        // Note that this performs writes to the storage, so we should do that here, rather than
740        // when working with the result.
741        let mls_message = group.content_to_mls_message(create_commit_result.commit, provider)?;
742
743        Ok(CommitMessageBundle {
744            version: group.version(),
745            commit: mls_message,
746            welcome: create_commit_result.welcome_option,
747            group_info: create_commit_result.group_info,
748        })
749    }
750}

752/// Contains the messages that are produced by committing. The messages can be accessed individually
753/// using getters or through the [`IntoIterator`] interface.
754#[derive(Debug, Clone)]
755pub struct CommitMessageBundle {
756    version: ProtocolVersion,
757    commit: MlsMessageOut,
758    welcome: Option<Welcome>,
759    group_info: Option<GroupInfo>,
760}
761
762#[cfg(test)]
763impl CommitMessageBundle {
764    pub fn new(
765        version: ProtocolVersion,
766        commit: MlsMessageOut,
767        welcome: Option<Welcome>,
768        group_info: Option<GroupInfo>,
769    ) -> Self {
770        Self {
771            version,
772            commit,
773            welcome,
774            group_info,
775        }
776    }
777}

779impl CommitMessageBundle {
780    // borrowed getters
781
782    /// Gets a the Commit messsage. For owned version, see [`Self::into_commit`].
783    pub fn commit(&self) -> &MlsMessageOut {
784        &self.commit
785    }
786
787    /// Gets a the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
788    /// For owned version, see [`Self::into_welcome`].
789    pub fn welcome(&self) -> Option<&Welcome> {
790        self.welcome.as_ref()
791    }
792
793    /// Gets a the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
794    /// Performs a copy of the Welcome. For owned version, see [`Self::into_welcome_msg`].
795    pub fn to_welcome_msg(&self) -> Option<MlsMessageOut> {
796        self.welcome
797            .as_ref()
798            .map(|welcome| MlsMessageOut::from_welcome(welcome.clone(), self.version))
799    }
800
801    /// Gets a the GroupInfo message. Only [`Some`] if new clients have been added or the group
802    /// configuration has `use_ratchet_tree_extension` set.
803    /// For owned version, see [`Self::into_group_info`].
804    pub fn group_info(&self) -> Option<&GroupInfo> {
805        self.group_info.as_ref()
806    }
807
808    /// Gets all three messages, some of which optional. For owned version, see
809    /// [`Self::into_contents`].
810    pub fn contents(&self) -> (&MlsMessageOut, Option<&Welcome>, Option<&GroupInfo>) {
811        (
812            &self.commit,
813            self.welcome.as_ref(),
814            self.group_info.as_ref(),
815        )
816    }
817
818    // owned getters
819    /// Gets a the Commit messsage. This method consumes the [`CommitMessageBundle`]. For a borrowed
820    /// version see [`Self::commit`].
821    pub fn into_commit(self) -> MlsMessageOut {
822        self.commit
823    }
824
825    /// Gets a the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
826    /// This method consumes the [`CommitMessageBundle`]. For a borrowed version see
827    /// [`Self::welcome`].
828    pub fn into_welcome(self) -> Option<Welcome> {
829        self.welcome
830    }
831
832    /// Gets a the Welcome messsage. Only [`Some`] if new clients have been added in the commit.
833    /// For a borrowed version, see [`Self::to_welcome_msg`].
834    pub fn into_welcome_msg(self) -> Option<MlsMessageOut> {
835        self.welcome
836            .map(|welcome| MlsMessageOut::from_welcome(welcome, self.version))
837    }
838
839    /// Gets a the GroupInfo message. Only [`Some`] if new clients have been added or the group
840    /// configuration has `use_ratchet_tree_extension` set.
841    /// This method consumes the [`CommitMessageBundle`]. For a borrowed version see
842    /// [`Self::group_info`].
843    pub fn into_group_info(self) -> Option<GroupInfo> {
844        self.group_info
845    }
846
847    /// Gets a the GroupInfo messsage. Only [`Some`] if new clients have been added in the commit.
848    pub fn into_group_info_msg(self) -> Option<MlsMessageOut> {
849        self.group_info.map(|group_info| group_info.into())
850    }
851
852    /// Gets all three messages, some of which optional. This method consumes the
853    /// [`CommitMessageBundle`]. For a borrowed version see [`Self::contents`].
854    pub fn into_contents(self) -> (MlsMessageOut, Option<Welcome>, Option<GroupInfo>) {
855        (self.commit, self.welcome, self.group_info)
856    }
857
858    /// Gets all three messages, some of which optional, as [`MlsMessageOut`].
859    /// This method consumes the [`CommitMessageBundle`].
860    pub fn into_messages(self) -> (MlsMessageOut, Option<MlsMessageOut>, Option<MlsMessageOut>) {
861        (
862            self.commit,
863            self.welcome
864                .map(|welcome| MlsMessageOut::from_welcome(welcome, self.version)),
865            self.group_info.map(|group_info| group_info.into()),
866        )
867    }
868}
869
870impl IntoIterator for CommitMessageBundle {
871    type Item = MlsMessageOut;
872
873    type IntoIter = core::iter::Chain<
874        core::iter::Chain<
875            core::option::IntoIter<MlsMessageOut>,
876            core::option::IntoIter<MlsMessageOut>,
877        >,
878        core::option::IntoIter<MlsMessageOut>,
879    >;
880
881    fn into_iter(self) -> Self::IntoIter {
882        let welcome = self.to_welcome_msg();
883        let group_info = self.group_info.map(|group_info| group_info.into());
884
885        Some(self.commit)
886            .into_iter()
887            .chain(welcome)
888            .chain(group_info)
889    }
